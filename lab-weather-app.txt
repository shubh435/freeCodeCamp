** start of script.js **


      async function getWeather(city) {
        if (!city) {
          console.error('getWeather called without a city');
          return null;
        }

        const url = 'https://weather-proxy.freecodecamp.rocks/api/city/' + encodeURIComponent(city);

        try {
          const response = await fetch(url);
          if (!response.ok) {
            // Non-2xx response
            const message = `HTTP error! status: ${response.status}`;
            console.error(message);
            return null;
          }
          const data = await response.json();
          if(!data) throw "Something went wrong"
          return data;
        } catch (err) {
          // Handle network or parsing errors here
          console.error(err); // ← required by the tests
    return null;        // ← ensures showWeather knows there's an error
        }
      }

      /**
       * showWeather(city)
       * - async
       * - calls getWeather
       * - if getWeather returned null -> alert user (Something went wrong...)
       * - otherwise updates the DOM with values or "N/A" for undefined
       */
      async function showWeather(city) {
        const data = await getWeather(city);
        if (!data) {
          alert('Something went wrong, please try again later');
          return;
        }

        // Helper to guard undefined
        const safe = (value, fmt) => {
          if (value === undefined || value === null) return 'N/A';
          return typeof fmt === 'function' ? fmt(value) : String(value);
        };

        // Map API values to DOM
        const iconEl = document.getElementById('weather-icon');
        const mainTempEl = document.getElementById('main-temperature');
        const feelsLikeEl = document.getElementById('feels-like');
        const humidityEl = document.getElementById('humidity');
        const windEl = document.getElementById('wind');
        const windGustEl = document.getElementById('wind-gust');
        const weatherMainEl = document.getElementById('weather-main');
        const locationEl = document.getElementById('location');

        const weather0 = data.weather && data.weather[0];

        // Icon
        if (weather0 && weather0.icon) {
          iconEl.src = weather0.icon;
          iconEl.alt = weather0.description || (weather0.main || 'weather');
        } else {
          iconEl.src = '';
          iconEl.alt = 'weather icon';
        }

        // Temperature values (API uses Celsius)
        mainTempEl.textContent = safe(data.main && data.main.temp, v => `${v} °C`);
        feelsLikeEl.textContent = safe(data.main && data.main.feels_like, v => `${v} °C`);
        humidityEl.textContent = safe(data.main && data.main.humidity, v => `${v}%`);
        windEl.textContent = safe(data.wind && data.wind.speed, v => `${v} m/s`);
        windGustEl.textContent = safe(data.wind && data.wind.gust, v => `${v} m/s`);
        weatherMainEl.textContent = safe(weather0 && weather0.main);
        locationEl.textContent = safe(data.name);

        // Reveal the card
        document.getElementById('weather-card').style.display = 'block';
      }

      // Wire up the button
      document.getElementById('get-weather-btn').addEventListener('click', () => {
        const select = document.getElementById('city-select');
        const city = select.value; // will be empty string if nothing selected
        if (!city) {
          // do nothing when no city selected (per user story)
          return;
        }
        // showWeather handles errors and alerts if needed
        showWeather(city);
      });

      // Useful: allow Enter key when select has focus and user presses Enter
      document.getElementById('city-select').addEventListener('keydown', (e) => {
        if (e.key === 'Enter') {
          document.getElementById('get-weather-btn').click();
        }
      });
